Die Klasse Verweisboxhat einen Inhalt und zwei Zeiger unten und oben. 

 class VerweisBox:
   def __init__(self, inhalt, unten=None, oben=None):
      self.inhalt = inhalt
      self.unten = unten
      self.oben = oben
      if unten is not None: self.unten.oben = self
      if oben  is not None: self.oben.unten = self

  def __str__(self):
      return self.inhalt
