\question[4] Implementiere die Funktion.
\begin{lstlisting}
def gemeinsameZeichen(s1, s2):
    '''
    s1, s2: Strings
    returns: eine sortierte Liste der Zeichen, die in beiden Strings
        vorkommen

    Beispiel:
    >>> s1 = 'ahooobbhhda'
    >>> s2 = 'hoollke'
    >>> gemeinsameZeichen(s1, s2)
    ['h', 'o']
    '''
\end{lstlisting}
\begin{solutionbox}{4cm}
\begin{lstlisting}
def gemeinsameZeichen(s1, s2):
    return sorted(set(s1) & set(s2))
\end{lstlisting}
\end{solutionbox}
