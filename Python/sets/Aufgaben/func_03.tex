\question[4] Implementiere die Funktion.
\begin{lstlisting}
def anzahlZweier(s):
    '''
    s: String
    returns: Anzahl der verschiedenen Teilstrings mit LÃ¤nge 2 in s

    Beispiel:
    >>> s = 'abababc'
    >>> anzahlZweier(s)
    3
    '''
\end{lstlisting}
\begin{solutionbox}{6cm}
\begin{lstlisting}
def anzahlZweier(s):
    tmp = set()
    for i in range(len(s)-1):
        tmp.add(s[i:i+2])
    return len(tmp)

\end{lstlisting}
\end{solutionbox}
