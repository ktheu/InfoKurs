\question[3] Implementiere die Funktion \texttt{bereich}.
\begin{lstlisting}
def bereich(u,v,x1,x2):
    '''
    u, v: sets mit ganzen Zahlen
    x1, x2: ganze Zahlen
    returns: set mit allen Zahlen k aus u oder v, f√ºr die
        x1 <= k <= x2 gilt.

    Beispiel:
    >>> u = {-1,3,5,17,9}
    >>> v = {2,14,6,28,10}
    >>> bereich(u,v,5,8)
    {5, 6}
    '''
\end{lstlisting}
\begin{solutionbox}{6cm}
\begin{lstlisting}
def bereich(u,v,x1,x2):
    tmp = set()
    for x in (u | v):
        if x1 <= x <= x2:
            tmp.add(x)
    return tmp
\end{lstlisting}
\end{solutionbox}
