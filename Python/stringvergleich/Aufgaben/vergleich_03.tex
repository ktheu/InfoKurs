\question[3] Implementiere die Funktion.
\begin{lstlisting}
def anzahlGrossbuchstaben(s):
    '''
    s: String
    returns: Anzahl Grossbuchstaben A-Z in s
    '''
\end{lstlisting}
\begin{solutionbox}{8cm}
\begin{lstlisting}
def anzahlGrossbuchstaben(s):
    zaehl = 0
    for c in s:
        if c.isupper():
            zaehl += 1
    return zaehl
\end{lstlisting}
\end{solutionbox}
