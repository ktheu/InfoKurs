\question[3] Implementiere die Funktion.

\begin{lstlisting}
def anzahlZiffern(s):
    '''
    s: String
    returns: int, Anzahl der Ziffern in s
    '''
\end{lstlisting}
\begin{solutionbox}{8cm}
\begin{lstlisting}
def anzahlZiffern(s):
    zaehl = 0
    for c in s:
        if '0' <= c <= '9': zaehl += 1
    return zaehl
\end{lstlisting}
\end{solutionbox}
