\question[4] Implementiere die Funktion.

\begin{lstlisting}
def mehrKleine(s):
    '''
    s: String
    returns: True, wenn in s mehr Zeichen a-z als Zeichen A-Z vorkommen
    '''
\end{lstlisting}
\begin{solutionbox}{8cm}
\begin{lstlisting}
def mehrKleine(s):
    summeKlein = 0
    summeGross = 0
    for c in s:
        if 'a' <= c <= 'z': summeKlein += 1
        elif 'A' <= c <= 'Z': summeGross += 1
    return summeKlein > summeGross
\end{lstlisting}
\end{solutionbox}
