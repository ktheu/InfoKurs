import turtle as t

def square(x,y,k):
    t.seth(0)
    t.up()
    t.setpos(x,y)
    t.down()
    for i in range(4):
        t.fd(k)
        t.left(90)
        
t.reset()
t.hideturtle()
t.tracer(0,0)

t.bgcolor('yellow')
t.pencolor('red')
t.width(5)
t.setheading(30)
t.fd(80)
t.update()

square(50,100,30)
t.fillcolor('green')
t.begin_fill()
square(50,-100,40)
t.end_fill()
t.color('brown','cyan')    # pencolor + fillcolor
t.begin_fill()
square(-50,-100,40)
t.end_fill()
       
t.update()

Events, mainloop, colormode, rgb-Farben

	
import turtle as t
import random

def square(x,y,k,farbe):
    t.seth(0)
    t.up()
    t.setpos(x,y)
    t.down()
    t.color(farbe)
    t.begin_fill()
    for i in range(4):
        t.fd(k)
        t.left(90)
    t.end_fill()

def r_key():
    t.reset()
    t.hideturtle()
    
def leftmouse(x,y):
    r = random.randint(0,255)
    g = random.randint(0,255)
    b = random.randint(0,255)
    square(x,y,random.randint(80,200),((r,g,b)))
    t.update()
           
t.reset()
t.hideturtle()
t.tracer(0,0)
t.colormode(255)
t.onscreenclick(leftmouse,btn=1)
t.onkey(r_key,'r')
t.listen()
t.update()
t.mainloop()
