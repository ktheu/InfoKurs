\question[4] Was erscheint auf der Konsole?
\begin{lstlisting}
for i in range(2,7):
    if i % 3 $==$ 4:
        break
    print(i)
print("Ende1")
j = 10
while j > 0:
    j = j - 3
    if j % 2 $==$ 0:
        continue
    print(j)
    j-=1
print("Ende2")
