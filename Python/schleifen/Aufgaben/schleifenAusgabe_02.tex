\question[4] Was erscheint auf der Konsole?
\begin{lstlisting}
zaehl = 0
zahl = 10
while zaehl < 10:
    print(zahl,end=' ')
    if zaehl % 2 $==$ 0:
        zahl+=5
    else:
        zahl+=3
    zaehl+=1
\end{lstlisting}
\begin{solutionbox}{1cm}
\texttt{10 15 18 23 26 31 34 39 42 47 }
\end{solutionbox}
