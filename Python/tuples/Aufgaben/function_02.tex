\question[4]
Implementiere die Funktion \texttt{maxchar}
\begin{lstlisting}
def maxchar(s):
    '''
    s: nicht leerer String
    returns: Tupel (c,k)
       c: Zeichen, das am hÃ¤ufigsten in s vorkommt, bei mehreren Zeichen das erste.
       k: Anzahl der Vorkommen von c

    Beispiel:
    >>> s = 'abbaacccccaa'
    >>> maxchar(s)
    ('a', 5)
    '''
\end{lstlisting}
\begin{solutionbox}{9cm}
\begin{lstlisting}
def maxchar(s):
    maxZeichen = s[0]
    maxCount = s.count(maxZeichen)
    for c in s:
        if s.count(c) > maxCount:
            maxZeichen = c
            maxCount = s.count(c)
    return maxZeichen, maxCount
\end{lstlisting}
\end{solutionbox}
