"""
12 Zeichen ab Codepoint 0x2654 in UTF-8 Kodierung in eine Datei schreiben.
"""
import codecs

f = codecs.open("unicode.txt", "w", "utf-8")
f.write('\ufeff') # utf-8 BOM kodierung: EF BB BF
k = 0x2654
for i in range(12):
    message = chr(k) + '\n'
    f.write(message)
    k = k + 1
f.close()
